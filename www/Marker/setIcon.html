var mapDiv = document.getElementById("map_canvas");
var map = plugin.google.maps.Map.getMap(mapDiv);

// Show a virtual dialog (loader.js)
showVirtualDialog(mapDiv, "Click on the markers!");

// Add markers
var markers = [];
var bounds = [];
POINTS.forEach(function(markerOptions) {
  var marker = map.addMarker(markerOptions);
  markers.push(marker);
  bounds.push(markerOptions.position);

  // If you click on a marker, the marker's icon will be changed.
  marker.on(plugin.google.maps.event.MARKER_CLICK, onMarkerClick);
  marker.on(plugin.google.maps.event.INFO_CLICK, onMarkerClick);
});

map.moveCamera({
  target: bounds
});


function onMarkerClick() {
  var marker = this;
  var iconData = marker.get("iconData");

  // Change the marker icon
  marker.setIcon(iconData);
}
