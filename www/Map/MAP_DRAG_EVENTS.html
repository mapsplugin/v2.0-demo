event: MAP_DRAG events

    • The MAP_DRAG_START event is fired when you drag the map with gesture.
    • The MAP_DRAG event is fired while you are dragging the map.
    • The MAP_DRAG_END event is fired when the dragging is stopped.
    map status : 
    var label = document.getElementById("label");
var div = document.getElementById("map_canvas");
var map = plugin.google.maps.Map.getMap(div);

// Catch all camera events
map.on(plugin.google.maps.event.MAP_DRAG_START, onMapDragEvents.bind(map, plugin.google.maps.event.MAP_DRAG_START));
map.on(plugin.google.maps.event.MAP_DRAG, onMapDragEvents.bind(map, plugin.google.maps.event.MAP_DRAG));
map.on(plugin.google.maps.event.MAP_DRAG_END, onMapDragEvents.bind(map, plugin.google.maps.event.MAP_DRAG_END));

function onMapDragEvents(eventName) {
  label.innerText = eventName;
}
