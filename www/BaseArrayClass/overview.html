baseArray.mapSeries()

    The same as map but runs only a single async operation at a time. 

    var idx = 1;
var baseArray = new plugin.google.maps.BaseArrayClass([1, 2, 3, 4, 5]);
baseArray.mapSeries(function(value, next) {

  setTimeout(function() {
    next(value + (idx++));
  }, 10);

}, function(results) {
  console.log(results);  // [2, 4, 6, 8, 10]
});
